void HAL_UART_RxCpltCallback(UART_HandleTypeDef *huart) {
    char confirm_msg[120] = {0};  // PC 확인 메시지 버퍼
    //char at_command[] = "AT+GID:1\r\n";
    if (huart->Instance == USART2) {  // PC → STM32
            if (pc_rx_index < sizeof(pc_rx_buffer) - 1) {
                pc_rx_buffer[pc_rx_index++] = rxByte;  // 버퍼에 저장

                // '\n'을 받으면 Wi-SUN 모듈로 "AT+GID?" 전송
                if (rxByte == '\n') {
                	if (pc_rx_index >= 2 && pc_rx_buffer[pc_rx_index - 2] == '\r') {
                	                    pc_rx_index -= 2;  // '\r\n' 제거
                	                }
                    // PC로 확인 메시지 전송 (Received)
                    snprintf(confirm_msg, sizeof(confirm_msg), "Received: %s", pc_rx_buffer);

                    HAL_UART_Transmit(&huart2, (uint8_t *)confirm_msg, strlen(confirm_msg), HAL_MAX_DELAY);
                    pc_rx_buffer[pc_rx_index++] = '\r';
					pc_rx_buffer[pc_rx_index++] = '\n';

					HAL_UART_Transmit(&huart1, (uint8_t *)pc_rx_buffer, strlen((char *)pc_rx_buffer), HAL_MAX_DELAY);

                    // Wi-SUN 모듈 응답 수신 시작
                    wisun_rx_index = 0;
                    HAL_UART_Receive_IT(&huart1, &rxByte, 1);

                    // 버퍼 초기화
                    memset(pc_rx_buffer, 0, sizeof(pc_rx_buffer));
                    pc_rx_index = 0;
                }
            } else {  // 버퍼 오버플로우 방지
                pc_rx_index = 0;
                memset(pc_rx_buffer, 0, sizeof(pc_rx_buffer));
            }
            // PC 데이터 계속 수신
            HAL_UART_Receive_IT(&huart2, &rxByte, 1);
        }

    else if (huart->Instance == USART1) {  // Wi-SUN → STM32
    	if (wisun_rx_index < sizeof(wisun_rx_buffer) - 1) {
    	            wisun_rx_buffer[wisun_rx_index++] = rxByte;

    	            // '\n'을 받으면 PC로 전송
    	            if (rxByte == '\r' || rxByte == '\n') {
    	            	char debug_msg[150];
						snprintf(debug_msg, sizeof(debug_msg), "[Wi-SUN Response] %s", wisun_rx_buffer);

						HAL_UART_Transmit(&huart2, (uint8_t *)debug_msg, strlen(debug_msg), HAL_MAX_DELAY);

						// 추가 디버깅
						//HAL_UART_Transmit(&huart2, (uint8_t *)"Debug: Sent to PC\n", 18, HAL_MAX_DELAY);

						// 버퍼 초기화
						memset(wisun_rx_buffer, 0, sizeof(wisun_rx_buffer));
						wisun_rx_index = 0;
					}
    	        } else {
    	            wisun_rx_index = 0;
    	            memset(wisun_rx_buffer, 0, sizeof(wisun_rx_buffer));
    	        }

    	        HAL_UART_Receive_IT(&huart1, &rxByte, 1);
    	    }
}

void Send_Broadcast_Command(uint8_t *data, uint8_t length) {
    if (length > 90) return;  // 최대 길이 제한 (패킷 크기 초과 방지)

    uint8_t packet[100] = {0};
    uint8_t checksum = 0;

    packet[0] = 0x02;   // STX
    packet[1] = 0xAA;   // SIG1
    packet[2] = 0xAB;   // SIG2
    packet[3] = length + 2;  // LEN (데이터 길이 + 2 (체크섬 + ETX 포함))
    packet[4] = 0x00;   // TMID (Broadcast 대상)
    packet[5] = 0x00;

    memcpy(&packet[6], data, length);  // 데이터 복사

    // Checksum 계산 (SIG1부터 TMID + DATA까지 XOR 연산)
    for (int i = 1; i < 6 + length; i++) {
        checksum ^= packet[i];
    }
    packet[6 + length] = checksum;  // 체크섬 추가
    packet[7 + length] = 0x03;  // ETX

    // Wi-SUN 모듈로 패킷 전송
    HAL_UART_Transmit(&huart1, packet, 8 + length, HAL_MAX_DELAY);
}
